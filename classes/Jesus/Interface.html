    def status
      begin
        statuses = server.status
      rescue
        return nil
      end
      groups = Hash.new
      
      statuses.each do |name, status|
        g = status[:group] || ''
        groups[g] ||= Hash.new
        groups[g][name] = status
      end
      groups
    end
